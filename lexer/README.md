## 各类单词的词法规则
1. digit -> [0-9]
2. letter -> [A-Za-z]
3. identifier -> [letter|_][letter|digit|_]*
4. keyword -> **'int'**|**'float'**|**'bool'**|**'record'**|**'if'**|**'else'**|**'do'**|**'while'**
5. arithmetic_operator -> '+' | '-' | '*' | '/' | '%' 
6. relation_operator -> '!=' | '>' | '<' | '>=' | '<='
7. logical_operator -> '&' | '|' | '^' | '&&' | '||' | '!'
8. delimiters -> '=' | ';' | '[' | ']' | '(' | ')' | '{' | '}'
9.  number -> digit+(.digit+)?((e|E)(+|-)?digit+)
10. comment -> /\*([^\*]*\*)*\*/(其中"."匹配任意字符)
